\begin{columns}[T]

% ── Left: control flow diagram ─────────────────────────────────────────────────
\begin{column}{0.44\textwidth}
\begin{tikzpicture}[node distance=4.5mm, scale=0.95, transform shape]

  \node[mitigate, fill=darkblue!8, draw=accent, text width=5cm]
    (req) {\textbf{Incoming Request}};

  \node[mitigate, fill=danger!12, draw=danger, text width=5cm, below=of req]
    (check1) {\textbf{Both CL + TE present?} \hfill \textcolor{danger}{→ 400}\\
    \textcolor{gray}{\tiny\texttt{map "\$http\_te:\$http\_cl" \$reject \{ "~\^{}[..][..].+" 1; \}}}};

  \node[mitigate, text width=5cm, below=of check1]
    (strip) {\textbf{Strip Transfer-Encoding}\\
    \textcolor{gray}{\tiny\texttt{proxy\_set\_header Transfer-Encoding ""}}\\
    \textcolor{darkblue!60}{\scriptsize → backend never sees TE header}};

  \node[mitigate, text width=5cm, below=of strip]
    (buf) {\textbf{Buffer Full Request}\\
    \textcolor{gray}{\tiny\texttt{proxy\_request\_buffering on}}\\
    \textcolor{darkblue!60}{\scriptsize → eliminates partial-write races}};

  \node[mitigate, text width=5cm, below=of buf]
    (conn) {\textbf{Force Connection: close}\\
    \textcolor{gray}{\tiny\texttt{proxy\_set\_header Connection "close"}}\\
    \textcolor{darkblue!60}{\scriptsize → destroys any lingering socket bytes}};

  \node[mitigate, fill=success!12, draw=success, text width=5cm, below=of conn]
    (h2) {\textbf{Upgrade to HTTP/2}\\
    \textcolor{darkblue!60}{\scriptsize binary framing → neutralises CL/TE ambiguity entirely}};

  \draw[greenarrow] (req.south) -- (check1.north);
  \draw[redarrow, dashed] (check1.east) -- ++(0.5,0) node[right, font=\tiny, text=danger]{\textbf{Reject 400}};
  \draw[greenarrow] (check1.south) -- (strip.north);
  \draw[greenarrow] (strip.south) -- (buf.north);
  \draw[greenarrow] (buf.south) -- (conn.north);
  \draw[greenarrow] (conn.south) -- (h2.north);

\end{tikzpicture}
\end{column}

\begin{block}{\small Mitigation Layer Summary}
\begin{tabular}{@{}p{1.6cm} p{4.5cm}@{}}
\toprule
\textbf{Layer} & \textbf{Action} \\
\midrule
\textbf{Nginx} & \texttt{underscores\_in\_headers on} (expose header for map); reject dual-length → 400; strip TE; buffer; close \\[2pt]
\textbf{Varnish} & Reject dual-length at cache edge; \texttt{unset bereq.http.TE}; \texttt{Connection: close}; never cache 5xx \\[2pt]
\textbf{Backend} & Drain \texttt{Content-Length} body on non-chunked requests to prevent socket corruption \\
\bottomrule
\end{tabular}
\end{block}
